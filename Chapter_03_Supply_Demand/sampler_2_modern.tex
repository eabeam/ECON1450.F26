\begin{frame}{Knowledge Check}
\phantomsection\label{knowledge-check}
\textbf{Question:} If garden gnomes regain popularity, what happens to
equilibrium price and quantity?

\vskip1em

\begin{enumerate}
[a)]
\item
  Price ↑, Quantity ↑
\item
  Price ↑, Quantity ↓
\item
  Price ↓, Quantity ↑
\item
  Price ↓, Quantity ↓
\end{enumerate}

\vskip2em

\pause

\begin{block}{Answer: (a) Price ↑, Quantity ↑}
\phantomsection\label{answer-a-price-quantity}
\textbf{Reasoning:} Tastes shift demand right → higher willingness to
pay at every quantity → new equilibrium at higher price and quantity
\end{block}
\end{frame}
